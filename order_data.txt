import 'dart:convert';

class OrderData {
  OrderData(
      {this.orderId,
      this.destinationAddress,
      this.pickupAddress,
      this.date,
      this.orderTotal,
      this.status = 'Completed'});

  final String? orderId;
  final String? destinationAddress;
  final String? pickupAddress;
  final DateTime? date;
  final double? orderTotal;
  final String? status;

  Map<String, dynamic> toMap() {
    return <String, dynamic>{
      'orderId': orderId,
      'destinationAddress': destinationAddress,
      'pickupAddress': pickupAddress,
      'date': date?.millisecondsSinceEpoch,
      'orderTotal': orderTotal,
      'status': status,
    };
  }

  factory OrderData.fromMap(Map<String, dynamic> map) {
    return OrderData(
      orderId: map['orderId'] != null ? map['orderId'] as String : null,
      destinationAddress: map['destinationAddress'] != null ? map['destinationAddress'] as String : null,
      pickupAddress: map['pickupAddress'] != null ? map['pickupAddress'] as String : null,
      date: map['date'] != null ? DateTime.fromMillisecondsSinceEpoch(map['date'] as int) : null,
      orderTotal: map['orderTotal'] != null ? map['orderTotal'].toDouble() : null,
      status: map['status'] != null ? map['status'] as String : null,
    );
  }

  String toJson() => json.encode(toMap());

  factory OrderData.fromJson(String source) => OrderData.fromMap(json.decode(source) as Map<String, dynamic>);
}
